version: '3'
services:
  db:
    image: postgres
    ports:
      - "5432:5432"
    environment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: pasword
      POSTGRES_DB: authorization
      #POSTGRES_INITDB_ARGS: --encoding=UTF-8
  app:
    build: .
    ports:
      - "3000:3000"
    environment:
      POSTGRES_URL: postgresql://postgres:${POSTGRES_PASSWORD}@db:5432/${POSTGRES_DB}
    depends_on:
      - db
  migrate:
    build: .
    command: sh -c "prisma migrate up --create-db"
    depends_on:
      - db
